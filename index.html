Users interact through clean, intuitive public methods without needing to understand complex internal implementations or algorithms.

Encapsulation

Private fields + Public methods = Data Safety

  • private String name; private double x, y; ‚Üí Internal state hidden
  • private List<double[]> memory ‚Üí Training data protected
  • public void moveTo(...), learn(...) ‚Üí Only safe operations allowed
  • private void drawGrid() ‚Üí Internal helper, not exposed

No external code can corrupt the arm's position or brain's memory.

Abstraction

Hide complexity behind simple methods

  • arm.moveTo(x, y, angles) ‚Üí User doesn't see grid drawing or angle validation
  • brain.predict(x, y) ‚Üí Returns angles without exposing weighted average math
  • User calls pick() ‚Üí No need to know gripper mechanics

Clean API = Easy to use, hard to misuse.

Modularity

Three independent, focused classes

Change one class ‚Üí others unaffected.

import java.util.*;
public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        System.out.println("Welcome to RoboArmAI!");
        System.out.println("This robot arm can learn where to move using simple examples.\n");
        RoboticArm arm = new RoboticArm("RoboArm", 10, 10);
        SimpleBrain brain = new SimpleBrain();
        System.out.println("Let's train the arm with 3 examples (x, y, Œ∏1, Œ∏2):\n");
        for (int i = 1; i <= 3; i++) {
            System.out.print("Example " + i + " - Enter x: ");
            double x = sc.nextDouble();
            System.out.print("Enter y: ");
            double y = sc.nextDouble();
            System.out.print("Enter theta1 (angle1): ");
            double t1 = sc.nextDouble();
            System.out.print("Enter theta2 (angle2): ");
            double t2 = sc.nextDouble();
            brain.learn(x, y, t1, t2);
            System.out.println();
        }
        System.out.println("\nTraining complete! Now let's test the arm.\n");
        System.out.print("How many pick-and-place operations? ");
        int n = sc.nextInt();
        for (int i = 1; i <= n; i++) {
            System.out.println("\n--- Operation " + i + " ---");
            System.out.print("Enter pick point (x y): ");
            double px = sc.nextDouble();
            double py = sc.nextDouble();
            double[] pickAngles = brain.predict(px, py);
            arm.moveTo(px, py, pickAngles);
            arm.pick();
            System.out.print("Enter place point (x y): ");
            double qx = sc.nextDouble();
            double qy = sc.nextDouble();
            double[] placeAngles = brain.predict(qx, qy);
            arm.moveTo(qx, qy, placeAngles);
            arm.place();
        }
        System.out.println("\nAll operations complete!");
        sc.close();
    }
}
class RoboticArm {
    private String name;
    private int gridWidth, gridHeight;
    private double x, y;
   
    public RoboticArm(String name, int gridWidth, int gridHeight) {
        this.name = name;
        this.gridWidth = gridWidth;
        this.gridHeight = gridHeight;
        this.x = 0;
        this.y = 0;
    }
   
    public void moveTo(double x, double y, double[] angles) {
        this.x = x; this.y = y;
        System.out.printf("%s moving to (%.1f, %.1f)...\n", name, x, y);
        System.out.printf("Predicted Angles: Œ∏1=%.1f¬∞, Œ∏2=%.1f¬∞\n", angles[0], angles[1]);
        drawGrid();
    }
   
    public void pick() {
        System.out.printf("%s: Picking at (%.1f, %.1f)\n", name, x, y);
    }
   
    public void place() {
        System.out.printf("%s: Placing at (%.1f, %.1f)\n", name, x, y);
    }
   
    private void drawGrid() {
        System.out.println("\nWorkspace:");
        for (int row = gridHeight; row >= 0; row--) {
            for (int col = 0; col <= gridWidth; col++) {
                if (Math.round(x) == col && Math.round(y) == row)
                    System.out.print("* ");
                else if (row == 0 && col == 0)
                    System.out.print("O ");
                else
                    System.out.print(". ");
            }
            System.out.println();
        }
    }
}
class SimpleBrain {
    private List memory = new ArrayList<>();
   
    public void learn(double x, double y, double t1, double t2) {
        memory.add(new double[]{x, y, t1, t2});
        System.out.printf("Learned: (%.1f, %.1f) -> [%.1f¬∞, %.1f¬∞]\n", x, y, t1, t2);
    }
   
    public double[] predict(double x, double y) {
        if (memory.isEmpty()) return new double[]{0, 0};
       
        double totalWeight = 0, sumT1 = 0, sumT2 = 0;
        for (double[] data : memory) {
            double dx = data[0] - x;
            double dy = data[1] - y;
            double dist = Math.sqrt(dx * dx + dy * dy) + 0.001;
            double weight = 1 / dist;
            sumT1 += data[2] * weight;
            sumT2 += data[3] * weight;
            totalWeight += weight;
        }
       
        return new double[]{sumT1 / totalWeight, sumT2 / totalWeight};
    }
}
